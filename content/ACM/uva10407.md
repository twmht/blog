    int gcd(int a,int b){
        return b == 0?a:gcd(b,a%b);
        
    }
    int main(int argc, char const* argv[])
    {
        string line;

        while(getline(cin,line) && line[0] != '0'){
            istringstream stream(line);
            int n;
            vector<int> p;
            stream>>n;
            int a = n;
            while(stream>>n && n != 0){
                p.push_back(abs((n-a)));
            }
            int l = p.size();
            int ans = 0;
            if(l > 1){
            for(int i = 0;i <l;i++){
                ans = gcd(p[i],ans);
            }
            }
            else{
                ans = gcd(0,p[0]);
            }
            printf("%d\n",ans);
        }
        
        return 0;
    }
