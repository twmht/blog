prime

int main(int argc, char const* argv[])
{
    bool prime[2001]; 
    prime[0] = true;
    prime[1] = true;

    for(int i = 2;i<2001;i++){
        if(!prime[i]){
            for(int j = i*i;j<2001;j = j+i){
                prime[j] =true;
            }
        }
    }
    int p;

    char str[2001];
    while(scanf("%d",&p) != EOF){
        for(int i = 1;i <= p;i++){
            int ascii[256] = {0};
            scanf("%s",str);
            for(char *s = str;*s != '\0';s++){
                if(isalpha(*s)){
                    ascii[(int)*s]++;
                }else{
                    ascii[(int)*s]++;
                }
            }
            printf("Case %d: ",i);
            int count = 0;
            for(int j = 0;j<256;j++){
                if(!prime[ascii[j]]){
                    printf("%c",j);
                    count++;
                }
            }
            if(!count){
                printf("empty");
            }
            printf("\n");
        }

    }

    return 0;
}
